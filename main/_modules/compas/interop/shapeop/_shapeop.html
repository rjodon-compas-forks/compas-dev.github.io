def int_array(data):
    try:
        len(data[0])
    except TypeError:
        pass
    else:
        data = list(flatten(data))
    n = len(data)
    array = shapeopPython.intArray(n)
    for index, value in enumerate(data):
        array[index] = int(value)
    return array
